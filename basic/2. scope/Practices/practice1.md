নীচের কোডে কোথাও ভূল আছে কিনা সেটা খুজে বের করার চেষ্টা করুন। যদি থাকে তবে সেটা কেন হচ্ছে সেটা ব্যাখ্যা করুন।

```js

let myCar = 'Lamborghini Huracan Evo'; 

function changeCar() {
    let myCar = 'Ferrari SF90 Stradale';
    console.log(myCar);
}

let myCar = 'Ford GT';
console.log(myCar);

```
